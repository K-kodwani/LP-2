public class Calculator {

    public Decimal add(Decimal num1, Decimal num2) {
        return num1 + num2;
    }

    public Decimal subtract(Decimal num1, Decimal num2) {
        return num1 - num2;
    }

    public Decimal multiply(Decimal num1, Decimal num2) {
        return num1 * num2;
    }

    public Decimal divide(Decimal num1, Decimal num2) {
        if (num2 == 0) {
            throw new IllegalArgumentException('Cannot divide by zero');
        }
        return num1 / num2;
    }
}


Calculator calc = new Calculator();

Decimal addResult = calc.add(10.0, 5.0);
System.debug('Addition Result: ' + addResult);

Decimal subResult = calc.subtract(10.0, 5.0);
System.debug('Subtraction Result: ' + subResult);

Decimal mulResult = calc.multiply(10.0, 5.0);
System.debug('Multiplication Result: ' + mulResult);

Decimal divResult = calc.divide(10.0, 5.0);
System.debug('Division Result: ' + divResult);

try {
    calc.divide(10.0, 0.0);
} catch (Exception e) {
    System.debug('Caught Exception: ' + e.getMessage());
}
